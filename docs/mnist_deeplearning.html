#   ( C )
# Input(784) - Hidden_1(512) - Hidden_2(256) - Output(10) 
# Try with your own Architecture!
model = Sequential()
model.add(Dense(512, input_shape=(784,))) #  
model.add(Activation('relu')) #  
model.add(Dense(256)) # 
model.add(Activation('relu')) #  
model.add(Dense(10)) # 
model.add(Activation('softmax')) #       1   ( D )
model.summary()
